  relations {

      platform           = "WEDATA"
      resource_name      = "db_demo.1"
      description        = "DLC"
    }

    target {
      resource_unique_id = "fM8OgzE-AM2h4aaJmdXoPg"
      resource_type      = "TABLE"
      platform           = "WEDATA"
      resource_name      = "db_demo.2"

    processes {

      process_type     = "SCHEDULE_TASK"

    }
